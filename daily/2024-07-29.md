``` js
      // 이벤트 핸들러를 이용해서 문서의 모든 요소가 로드되면 실행
      window.onload() = function () { // 1
	      var demo = document.getElementById("demo");
	      console.log(demo);
      }

      window.onload() = () => { // 2
	      var demo = document.getElementById("demo");
	      console.log(demo);
      }

      window.addEventListener('load', function() { // 3
	      var demo = document.getElementById("demo");
	      console.log(demo);
      })
```
